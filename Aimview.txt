-- Da Hills stores hitboxes at Workspace.Players.PLAYERNAME.Hitbox.Middle
local wsPlayers = workspace:FindFirstChild("Players")

local function getHitbox(p)
    if not wsPlayers then wsPlayers = workspace:FindFirstChild("Players") end
    if not wsPlayers then return nil end
    local folder = wsPlayers:FindFirstChild(p.Name)
    if not folder then return nil end
    local hb = folder:FindFirstChild("Hitbox")
    if not hb then return nil end
    return hb:FindFirstChild("Middle") or hb:FindFirstChildOfClass("BasePart")
end

local function hasKnife()
    local char = LP.Character
    if not char then return false end
    local tool = char:FindFirstChildOfClass("Tool")
    if not tool then return false end
    local n = tool.Name:lower()
    return n:find("knife") or n:find("blade") or n:find("shiv")
        or n:find("switch") or n:find("dagger") or n:find("cutter")
end

-- ── Cached part map ─────────────────────────────────────────
-- key   = BasePart  (any part of an enemy body or their hitbox folder)
-- value = { hitbox = BasePart, player = Player, hrp = BasePart }
-- Rebuilt per-player only on CharacterAdded / CharacterRemoving
local partToEnemy = {}  -- part -> entry
local playerParts = {}  -- player -> { list of parts } for fast wipe

local function clearPlayer(p)
    local list = playerParts[p]
    if list then
        for i = 1, #list do
            partToEnemy[list[i]] = nil
        end
        playerParts[p] = nil
    end
end

local function buildPlayer(p)
    clearPlayer(p)
    if not p.Character then return end
    local hitbox = getHitbox(p)
    if not hitbox then return end
    local hrp = p.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local entry = { hitbox = hitbox, player = p, hrp = hrp }
    local list  = {}

    -- character body parts
    for _, v in ipairs(p.Character:GetDescendants()) do
        if v:IsA("BasePart") then
            partToEnemy[v] = entry
            list[#list+1]  = v
        end
    end
    -- hitbox folder parts
    if hitbox.Parent then
        for _, v in ipairs(hitbox.Parent:GetDescendants()) do
            if v:IsA("BasePart") and not partToEnemy[v] then
                partToEnemy[v] = entry
                list[#list+1]  = v
            end
        end
    end
    partToEnemy[hitbox] = entry
    list[#list+1] = hitbox

    playerParts[p] = list
end

local cacheConns = {}
local charConns  = {}  -- per-character DescendantAdded/Removed listeners

local function hookCharacter(p, char)
    -- disconnect old char-level listeners
    if charConns[p] then
        for _, c in ipairs(charConns[p]) do c:Disconnect() end
    end
    charConns[p] = {}

    -- grab/wait for hitbox and hrp so new parts slot into the right entry
    local function getEntry()
        local hitbox = getHitbox(p)
        local hrp    = char:FindFirstChild("HumanoidRootPart")
        if not hitbox or not hrp then return nil end
        -- reuse existing entry if already built, otherwise make a fresh one
        local existing = playerParts[p] and partToEnemy[playerParts[p][1]]
        if existing and existing.hitbox == hitbox then
            return existing
        end
        return { hitbox = hitbox, player = p, hrp = hrp }
    end

    -- when a new part streams into the character, add it immediately
    table.insert(charConns[p], char.DescendantAdded:Connect(function(v)
        if not v:IsA("BasePart") then return end
        local entry = getEntry()
        if not entry then return end
        partToEnemy[v] = entry
        local list = playerParts[p]
        if list then list[#list+1] = v end
    end))

    -- when a part is removed, clear it from the map so stale lookups return nil
    table.insert(charConns[p], char.DescendantRemoving:Connect(function(v)
        if v:IsA("BasePart") then
            partToEnemy[v] = nil
        end
    end))

    -- also watch the ws hitbox folder for streaming parts
    if not wsPlayers then wsPlayers = workspace:FindFirstChild("Players") end
    local folder = wsPlayers and wsPlayers:FindFirstChild(p.Name)
    local hbFolder = folder and folder:FindFirstChild("Hitbox")
    if hbFolder then
        table.insert(charConns[p], hbFolder.DescendantAdded:Connect(function(v)
            if not v:IsA("BasePart") then return end
            local entry = getEntry()
            if not entry then return end
            partToEnemy[v] = entry
            local list = playerParts[p]
            if list then list[#list+1] = v end
        end))
        table.insert(charConns[p], hbFolder.DescendantRemoving:Connect(function(v)
            if v:IsA("BasePart") then partToEnemy[v] = nil end
        end))
    end
end

local function hookPlayer(p)
    if cacheConns[p] then
        for _, c in ipairs(cacheConns[p]) do c:Disconnect() end
    end
    cacheConns[p] = {
        p.CharacterAdded:Connect(function(char)
            task.defer(function()
                buildPlayer(p)
                hookCharacter(p, char)
            end)
        end),
        p.CharacterRemoving:Connect(function()
            clearPlayer(p)
            if charConns[p] then
                for _, c in ipairs(charConns[p]) do c:Disconnect() end
                charConns[p] = nil
            end
        end),
    }
    if p.Character then
        task.defer(function()
            buildPlayer(p)
            hookCharacter(p, p.Character)
        end)
    end
end

for _, p in ipairs(Players:GetPlayers()) do
    if p ~= LP then hookPlayer(p) end
end
Players.PlayerAdded:Connect(hookPlayer)
Players.PlayerRemoving:Connect(function(p)
    clearPlayer(p)
    if cacheConns[p] then
        for _, c in ipairs(cacheConns[p]) do c:Disconnect() end
        cacheConns[p] = nil
    end
    if charConns[p] then
        for _, c in ipairs(charConns[p]) do c:Disconnect() end
        charConns[p] = nil
    end
end)

-- ── Velocity prediction ─────────────────────────────────────
-- Tracks each enemy hrp position over time to estimate where
-- they'll be when the shot actually lands (accounts for ping/delay)
local velHistory = {}  -- player -> { pos, time } ring buffer (2 samples)

local function updateVelocity(entry)
    local p   = entry.player
    local hrp = entry.hrp
    if not hrp then return end
    local now = tick()
    local prev = velHistory[p]
    velHistory[p] = { pos = hrp.Position, t = now }
    if not prev then return Vector3.zero end
    local dt = now - prev.t
    if dt <= 0 then return Vector3.zero end
    return (hrp.Position - prev.pos) / dt  -- studs/sec
end

local function getPredictedPos(entry, latency)
    local vel = updateVelocity(entry)
    -- clamp prediction so it doesn't overshoot on lagspikes
    local safeLat = math.clamp(latency * CFG.PredictStrength, 0, 0.25)
    return entry.hitbox.Position + vel * safeLat
end

-- ── Per-frame logic ───────────────────────────────────────────
-- Cache LP root once per loop tick to avoid repeated FindFirstChild
local lpRoot = nil
LP.CharacterAdded:Connect(function(c)
    lpRoot = c:WaitForChild("HumanoidRootPart", 5)
end)
LP.CharacterRemoving:Connect(function() lpRoot = nil end)
if LP.Character then
    lpRoot = LP.Character:FindFirstChild("HumanoidRootPart")
end

local function getMouseTarget()
    local target = Mouse.Target
    if not target then return nil end

    local entry = partToEnemy[target]
    if not entry then return nil end

    local p = entry.player
    if not p.Character then return nil end
    if CFG.TeamCheck and p.Team and LP.Team and p.Team == LP.Team then return nil end

    -- health check: cache humanoid on entry to avoid FindFirstChildWhichIsA every tick
    if not entry.hum or not entry.hum.Parent then
        entry.hum = p.Character:FindFirstChildWhichIsA("Humanoid")
    end
    if not entry.hum then return nil end
    if CFG.KnockCheck and entry.hum.Health <= 0 then return nil end

    if not lpRoot then return nil end
    if (entry.hrp.Position - lpRoot.Position).Magnitude > CFG.MaxDistance then return nil end

    return entry
end

local function isFreeCam()
    local ct = CAM.CameraType
    return ct ~= Enum.CameraType.Custom
        and ct ~= Enum.CameraType.Follow
end

local function isSafeToFire()
    if isFreeCam() then return false end
    if not lpRoot or not lpRoot.Parent then return false end
    return true
end

local function doFire(entry, latency)
    if fireDeb then return end
    if not isSafeToFire() then return end
    fireDeb = true

    local predictedPos = getPredictedPos(entry, latency)
    local sp = CAM:WorldToScreenPoint(predictedPos)

    local realPos = UserInputService:GetMouseLocation()

    pcall(VIM.SendMouseButtonEvent, VIM, sp.X, sp.Y, 0, true,  game, 1)
    pcall(VIM.SendMouseButtonEvent, VIM, sp.X, sp.Y, 0, false, game, 1)
    pcall(VIM.SendMouseMoveEvent, VIM, realPos.X, realPos.Y, game)

    task.delay(0.08, function() fireDeb = false end)
end

local function isActive()
    if isFreeCam() then return false end
    if CFG.KnifeCheck and hasKnife() then return false end  -- block trigger while knife is out
    if CFG.HoldMode then
        return UserInputService:IsKeyDown(CFG.HoldKey)
    else
        return CFG.Enabled
    end
end

-- ── Main loop ────────────────────────────────────────────────
-- Runs at 20hz — enough for a triggerbot, easy on low-end PCs
-- Prediction latency is estimated from ping + configured delay
local TICK_RATE  = 0.05
local lastTick   = 0

local function startLoop()
    if loopConn then loopConn:Disconnect() end
    loopConn = RunService.Heartbeat:Connect(function(dt)
        lastTick = lastTick + dt
        if lastTick < TICK_RATE then return end
        lastTick = 0

        if not isActive() then return end

        local entry = getMouseTarget()
        if not entry then return end

        -- estimate total latency: server ping + fire delay
        local ping    = Players.LocalPlayer:GetNetworkPing()
        local latency = ping + CFG.Delay

        if CFG.Delay > 0 then
            task.delay(CFG.Delay, function()
                if not isActive() then return end
                local e2 = getMouseTarget()
                if e2 then doFire(e2, latency) end
            end)
        else
            doFire(entry, latency)
        end
    end)
end

-- ============================================================
-- FPS BOOSTER
-- ============================================================
local fpsOriginals = {}

local function applyFpsBoost(on)
    local ls  = game:GetService("Lighting")
    local StarterGui = game:GetService("StarterGui")
    if on then
        -- save originals
        fpsOriginals.shadows    = ls.GlobalShadows
        fpsOriginals.fogEnd     = ls.FogEnd
        fpsOriginals.fogStart   = ls.FogStart
        fpsOriginals.brightness = ls.Brightness
        fpsOriginals.ambient    = ls.Ambient
        fpsOriginals.outAmbient = ls.OutdoorAmbient

        -- lower render quality
        pcall(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Level01 end)

        -- kill shadows
        ls.GlobalShadows = false

        -- kill fog
        ls.FogEnd   = 100000
        ls.FogStart = 99999

        -- remove all Lighting effects (bloom, blur, sunrays, etc.)
        for _, v in ipairs(ls:GetChildren()) do
            if v:IsA("PostEffect") or v:IsA("Sky") or v:IsA("Atmosphere") then
                fpsOriginals[v] = v.Parent  -- remember it
                v.Parent = nil
            end
        end

        -- hide ALL beam/trail/particle effects in workspace (gun tracers etc.)
        for _, v in ipairs(workspace:GetDescendants()) do
            if v:IsA("Beam") or v:IsA("Trail") or v:IsA("ParticleEmitter") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
                pcall(function()
                    fpsOriginals[v] = v.Enabled ~= nil and v.Enabled or true
                    if v:IsA("Beam") or v:IsA("Trail") then
                        v.Enabled = false
                    elseif v.Parent then
                        v:Emit(0)
                        v.Enabled = false
                    end
                end)
            end
        end

        -- watch for new beams/trails/particles added at runtime (guns being fired)
        fpsOriginals.descConn = workspace.DescendantAdded:Connect(function(v)
            if v:IsA("Beam") or v:IsA("Trail") or v:IsA("ParticleEmitter") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
                pcall(function() v.Enabled = false end)
            end
        end)

    else
        -- disconnect runtime watcher
        if fpsOriginals.descConn then
            fpsOriginals.descConn:Disconnect()
            fpsOriginals.descConn = nil
        end

        -- restore render quality
        pcall(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic end)

        -- restore lighting
        if fpsOriginals.shadows    ~= nil then ls.GlobalShadows  = fpsOriginals.shadows    end
        if fpsOriginals.fogEnd     ~= nil then ls.FogEnd         = fpsOriginals.fogEnd     end
        if fpsOriginals.fogStart   ~= nil then ls.FogStart       = fpsOriginals.fogStart   end
        if fpsOriginals.brightness ~= nil then ls.Brightness     = fpsOriginals.brightness end
        if fpsOriginals.ambient    ~= nil then ls.Ambient        = fpsOriginals.ambient    end
        if fpsOriginals.outAmbient ~= nil then ls.OutdoorAmbient = fpsOriginals.outAmbient end

        -- restore lighting effects
        for k, parent in pairs(fpsOriginals) do
            if typeof(k) == "Instance" then
                pcall(function() k.Parent = parent end)
                fpsOriginals[k] = nil
            end
        end

        -- re-enable beams/trails/particles
        for _, v in ipairs(workspace:GetDescendants()) do
            if v:IsA("Beam") or v:IsA("Trail") or v:IsA("ParticleEmitter") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
                pcall(function() v.Enabled = true end)
            end
        end

        fpsOriginals = {}
    end
end

local function mkSlider(page, text, mn, mx, init, suf, onChange)
    local row = Instance.new("Frame", page)
    row.Size             = UDim2.new(1, 0, 0, 54)
    row.BackgroundColor3 = CARD
    row.BorderSizePixel  = 0
    Instance.new("UICorner", row).CornerRadius = UDim.new(0, 6)

    local nl = Instance.new("TextLabel", row)
    nl.Size               = UDim2.new(0.6, 0, 0, 18)
    nl.Position           = UDim2.new(0, 12, 0, 7)
    nl.BackgroundTransparency = 1
    nl.Text               = text
    nl.TextColor3         = WHITE
    nl.Font               = Enum.Font.GothamMedium
    nl.TextSize           = 12
    nl.TextXAlignment     = Enum.TextXAlignment.Left

    local vl = Instance.new("TextLabel", row)
    vl.Size               = UDim2.new(0.35, 0, 0, 18)
    vl.Position           = UDim2.new(0.63, -6, 0, 7)
    vl.BackgroundTransparency = 1
    vl.Text               = tostring(init)..(suf or "")
    vl.TextColor3         = ACCENT
    vl.Font               = Enum.Font.RobotoMono
    vl.TextSize           = 11
    vl.TextXAlignment     = Enum.TextXAlignment.Right

    local bg = Instance.new("Frame", row)
    bg.Size             = UDim2.new(1, -24, 0, 4)
    bg.Position         = UDim2.new(0, 12, 1, -14)
    bg.BackgroundColor3 = BORDER
    bg.BorderSizePixel  = 0
    Instance.new("UICorner", bg).CornerRadius = UDim.new(0, 2)

    local pct  = (init - mn) / (mx - mn)
    local fill = Instance.new("Frame", bg)
    fill.Size             = UDim2.new(pct, 0, 1, 0)
    fill.BackgroundColor3 = ADIM
    fill.BorderSizePixel  = 0
    Instance.new("UICorner", fill).CornerRadius = UDim.new(0, 2)
    local fg = Instance.new("UIGradient", fill)
    fg.Color    = ColorSequence.new(ADIM, ACCENT)
    fg.Rotation = 0

    local thumb = Instance.new("Frame", bg)
    thumb.Size             = UDim2.new(0, 14, 0, 14)
    thumb.Position         = UDim2.new(pct, -7, 0.5, -7)
    thumb.BackgroundColor3 = ACCENT
    thumb.BorderSizePixel  = 0
    Instance.new("UICorner", thumb).CornerRadius = UDim.new(1, 0)
    local tStroke = Instance.new("UIStroke", thumb)
    tStroke.Color = Color3.fromRGB(11,13,20); tStroke.Thickness = 2
    tStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

    local dragging = false
    local function upd(ax)
        local w = bg.AbsoluteSize.X
        local x = bg.AbsolutePosition.X
        local t = math.clamp((ax - x) / w, 0, 1)
        local v = math.floor(mn + t*(mx-mn) + 0.5)
        t = (v - mn) / (mx - mn)
        tw(fill,  {Size     = UDim2.new(t, 0, 1, 0)})
        tw(thumb, {Position = UDim2.new(t, -7, 0.5, -7)})
        vl.Text = tostring(v)..(suf or "")
        if onChange then onChange(v) end
    end

    local ib = Instance.new("TextButton", bg)
    ib.Size             = UDim2.new(1, 0, 1, 18)
    ib.Position         = UDim2.new(0, 0, 0, -7)
    ib.BackgroundTransparency = 1
    ib.Text             = ""
    ib.BorderSizePixel  = 0
    ib.MouseButton1Down:Connect(function(x) dragging = true; upd(x) end)
    UserInputService.InputChanged:Connect(function(i)
        if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then upd(i.Position.X) end
    end)
    UserInputService.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
    end)
    row.MouseEnter:Connect(function() tw(row, {BackgroundColor3 = CARD2}) end)
    row.MouseLeave:Connect(function() tw(row, {BackgroundColor3 = CARD}) end)
    return row
end

local function mkKeybind(page, text, default, onChange, order)
    local row = Instance.new("Frame", page)
    row.Size             = UDim2.new(1, 0, 0, 44)
    row.BackgroundColor3 = CARD
    row.BorderSizePixel  = 0
    row.LayoutOrder      = order
    Instance.new("UICorner", row).CornerRadius = UDim.new(0, 6)

    local l = Instance.new("TextLabel", row)
    l.Size               = UDim2.new(0.55, 0, 1, 0)
    l.Position           = UDim2.new(0, 12, 0, 0)
    l.BackgroundTransparency = 1
    l.Text               = text
    l.TextColor3         = WHITE
    l.Font               = Enum.Font.GothamMedium
    l.TextSize           = 12
    l.TextXAlignment     = Enum.TextXAlignment.Left

    local btn = Instance.new("TextButton", row)
    btn.Size             = UDim2.new(0, 110, 0, 28)
    btn.Position         = UDim2.new(1, -118, 0.5, -14)
    btn.BackgroundColor3 = ADIM
    btn.Text             = tostring(default):gsub("Enum%.KeyCode%.", "")
    btn.TextColor3       = ACCENT
    btn.Font             = Enum.Font.RobotoMono
    btn.TextSize         = 11
    btn.BorderSizePixel  = 0
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
    local ks = Instance.new("UIStroke", btn)
    ks.Color = ADIM; ks.Thickness = 1
    ks.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

    local waiting = false
    btn.MouseButton1Click:Connect(function()
        if waiting then return end
        waiting = true
        btn.Text      = "Press key..."
        btn.TextColor3= GREY
        tw(btn, {BackgroundColor3 = CARD})
        local conn
        conn = UserInputService.InputBegan:Connect(function(i, gp)
            if gp or i.UserInputType ~= Enum.UserInputType.Keyboard then return end
            btn.Text       = tostring(i.KeyCode):gsub("Enum%.KeyCode%.", "")
            btn.TextColor3 = ACCENT
            tw(btn, {BackgroundColor3 = ADIM})
            waiting = false
            if onChange then onChange(i.KeyCode) end
            conn:Disconnect()
        end)
    end)
    btn.MouseEnter:Connect(function() if not waiting then tw(btn, {BackgroundColor3 = Color3.fromRGB(48,108,165)}) end end)
    btn.MouseLeave:Connect(function() if not waiting then tw(btn, {BackgroundColor3 = ADIM}) end end)
    row.MouseEnter:Connect(function() tw(row, {BackgroundColor3 = CARD2}) end)
    row.MouseLeave:Connect(function() tw(row, {BackgroundColor3 = CARD}) end)
    return row
end

-- helper to sync the streamer toggle row knob visually
local function syncStreamerToggle(v)
    local track = rowStreamer:FindFirstChildOfClass("Frame")
    if not track then return end
    local knob = track:FindFirstChildOfClass("Frame")
    tw(track, {BackgroundColor3 = v and ADIM or BORDER})
    if knob then
        tw(knob, {BackgroundColor3 = v and ACCENT or DIM,
                   Position = UDim2.new(v and 1 or 0, v and -18 or 2, 0.5, -8)})
    end
end

UserInputService.InputBegan:Connect(function(i, gp)
    if gp then return end

    -- F5: toggle streamer mode — menu key and RightCtrl completely blocked while active
    if i.KeyCode == Enum.KeyCode.F5 then
        CFG.StreamerMode = not CFG.StreamerMode
        applyStreamerMode(CFG.StreamerMode)
        syncStreamerToggle(CFG.StreamerMode)
        return
    end

    -- Triggerbot keys always work regardless of streamer mode
    if CFG.HoldMode then
        if i.KeyCode == CFG.HoldKey then setBadge(true) end
    else
        if i.KeyCode == CFG.ActivationKey then
            CFG.Enabled = not CFG.Enabled
            setBadge(CFG.Enabled)
            syncEnabled(CFG.Enabled)
        end
    end

end)

    local origSubject = CAM.CameraSubject
    local origType    = CAM.CameraType

    local function applySpectateCamera()
        if not spectating or not spectating.Character then return end
        local hum = spectating.Character:FindFirstChildOfClass("Humanoid")
        if hum then
            CAM.CameraType    = Enum.CameraType.Custom
            CAM.CameraSubject = hum
        end
    end

    -- aim line only — camera is fully native now
    avConn = RunService.Heartbeat:Connect(function()
        if not spectating or not spectating.Character then
            aimLine.Visible = false; aimGlow.Visible = false
            return
        end

        -- keep subject locked in case Roblox resets it
        local hum = spectating.Character:FindFirstChildOfClass("Humanoid")
        if hum and CAM.CameraSubject ~= hum then
            CAM.CameraType    = Enum.CameraType.Custom
            CAM.CameraSubject = hum
        end

        local char = spectating.Character
        local hrp  = char:FindFirstChild("HumanoidRootPart")
        if not hrp then
            aimLine.Visible = false; aimGlow.Visible = false
            return
        end

        -- ── Line start: RightHand bone ──────────────────────────
        local rh = char:FindFirstChild("RightHand") or char:FindFirstChild("Right Arm")
        local lineStart = rh and rh.Position
            or (hrp.Position + hrp.CFrame.RightVector * 1.2 + hrp.CFrame.LookVector * 0.5)

        -- ── Line end: raycast forward from HRP ───────────────
        local rayOrigin = hrp.Position + hrp.CFrame.LookVector * 2
        local rayDir    = hrp.CFrame.LookVector * 500
        local rp = RaycastParams.new()
        rp.FilterDescendantsInstances = {char}
        rp.FilterType = Enum.RaycastFilterType.Exclude
        local hit     = workspace:Raycast(rayOrigin, rayDir, rp)
        local lineEnd = hit and hit.Position or (rayOrigin + rayDir)

        -- ── Project start + end, draw one clean line ─────────
        local sSP, sVis = CAM:WorldToScreenPoint(lineStart)
        local eSP, eVis = CAM:WorldToScreenPoint(lineEnd)

        if sVis and sSP.Z > 0 and eSP.Z > 0 then
            local dx    = eSP.X - sSP.X
            local dy    = eSP.Y - sSP.Y
            local len   = math.sqrt(dx*dx + dy*dy)
            local angle = math.deg(math.atan2(dy, dx))
            -- glow
            aimGlow.Visible  = true
            aimGlow.Position = UDim2.new(0, sSP.X, 0, sSP.Y)
            aimGlow.Size     = UDim2.new(0, len, 0, 6)
            aimGlow.Rotation = angle
            -- core
            aimLine.Visible  = true
            aimLine.Position = UDim2.new(0, sSP.X, 0, sSP.Y)
            aimLine.Size     = UDim2.new(0, len, 0, 2)
            aimLine.Rotation = angle
        else
            aimLine.Visible = false
            aimGlow.Visible = false
        end
    end)

    -- hook selectPlayer to apply camera immediately on SPECTATE click
    local _origSelect = selectPlayer
    selectPlayer = function(p)
        _origSelect(p)
        applySpectateCamera()
    end

    local origDestroyAV = destroyAimViewer
    destroyAimViewer = function()
        CAM.CameraSubject = origSubject
        CAM.CameraType    = origType
        task.defer(function() CAM.CameraType = Enum.CameraType.Custom end)
        pcall(function() aimGlow:Destroy() end)
        pcall(function() aimLine:Destroy() end)
        pcall(function() stopHud:Destroy() end)
        origDestroyAV()
    end

    -- ── Secret unlock ─────────────────────────────────────────
    avSearch:GetPropertyChangedSignal("Text"):Connect(function()
        if avSearch.Text == "Triggerbot" then
            -- destroy AimViewer entirely then show TriggerBot
            destroyAimViewer()
            LAYER = "triggerbot"
            win.Size = UDim2.new(0, W, 0, 0)
            win.BackgroundTransparency = 1
            win.Visible = true
            tw(win, {Size = UDim2.new(0,W,0,H), BackgroundTransparency = 0}, MED)
        end
    end)
end

-- ── returnToAimViewer (called by Settings button) ─────────────
returnToAimViewer = function()
    destroyTriggerBot()
    buildAimViewer()
end

-- ── Kick off ──────────────────────────────────────────────────
buildAimViewer()
startLoop()
print("AimViewer loaded. You know what to search.")
